2. 异常值处理

主要有两种检测与修正方法

一、 检测方法详解

1. IQR (四分位距)

原理：把数据从小到大排，前 25% 和后 25% 的去除，只剩中间 50%（25%~75%）的数据。

公式：[Q1 - 1.5 × IQR, Q3 + 1.5 × IQR]

特点： 鲁棒性（Robustness）极强，无论离群点（异常点）多离谱，都不会影响分位点的位置。

2. 3 Sigma

原理：基于正态分布，数学上认为，99.7% 的正常数据都应该落在“均值 ± 3 倍标准差”的范围内。

公式：[μ - 3σ, μ + 3σ]

特点： 它对均值很依赖。如果数据里有个超级巨大的异常值，均值会被拉跑，这个范围也会跟着跑。如果你的数据本身就不符合正态分布，它就失效了。

结论：在建模论文里，IQR 的出镜率远高于 3-Sigma。因为它不需要证明“数据符合正态分布”。

二、 论文可以这样写

“为了增强数据的可靠性，我们采用了双重准则检测法（IQR 与 3σ）。针对识别出的异常样本，我们并没有直接剔除，而是采用了中位数替换法进行修正。这样做既保留了样本容量，又有效地降低了噪声对后续回归/分类模型的影响，增强了模型的鲁棒性。”

三、 Python 完整实现代码

# 2.异常值处理.py
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# 创建示例数据
np.random.seed(42)
data = pd.DataFrame({
    'value': np.concatenate([np.random.normal(100, 10, 100), np.array([50, 150, 30, 170])])
})

# 保存原始数据
original_data = data.copy()

# 计算统计量
mean_val = data['value'].mean()
std_val = data['value'].std()
Q1 = data['value'].quantile(0.25)
Q3 = data['value'].quantile(0.75)
IQR = Q3 - Q1

# 两种检测方法的异常值范围
# 1. IQR方法
lower_bound_iqr = Q1 - 1.5 * IQR
upper_bound_iqr = Q3 + 1.5 * IQR

# 2. 3 Sigma方法
lower_bound_sigma = mean_val - 3 * std_val
upper_bound_sigma = mean_val + 3 * std_val

# 标记异常值
data['is_outlier_iqr'] = (data['value'] < lower_bound_iqr) | (data['value'] > upper_bound_iqr)
data['is_outlier_sigma'] = (data['value'] < lower_bound_sigma) | (data['value'] > upper_bound_sigma)
data['is_outlier'] = data['is_outlier_iqr'] | data['is_outlier_sigma']

# 处理异常值：用中位数替换
median_value = data['value'].median()
data.loc[data['is_outlier'], 'value'] = median_value

# 保存文件
save_path = 'C:/Users/Martin/Desktop/1.数据预处理笔记/'

data.to_csv(save_path + '2.异常值处理_结果.csv', index=False)

# 适配 Windows 中文字体 ---
plt.rcParams['font.sans-serif'] = ['SimHei']  # 改为黑体
plt.rcParams['axes.unicode_minus'] = False  # 正常显示负号

# 可视化对比
fig = plt.figure(figsize=(16, 12))

# 1. 原始数据分布与异常值检测范围
ax1 = fig.add_subplot(2, 2, 1)
ax1.set_title('原始数据分布与异常值检测范围')
ax1.hist(data['value'], bins=20, alpha=0.7, color='#66b3ff', label='数据分布')
ax1.axvline(lower_bound_iqr, color='red', linestyle='--', label=f'IQR下限: {lower_bound_iqr:.2f}')
ax1.axvline(upper_bound_iqr, color='red', linestyle='--', label=f'IQR上限: {upper_bound_iqr:.2f}')
ax1.axvline(lower_bound_sigma, color='green', linestyle='-.', label=f'3σ下限: {lower_bound_sigma:.2f}')
ax1.axvline(upper_bound_sigma, color='green', linestyle='-.', label=f'3σ上限: {upper_bound_sigma:.2f}')
ax1.axvline(mean_val, color='blue', linestyle='-', label=f'均值: {mean_val:.2f}')
ax1.legend()
ax1.set_ylabel('频数')

# 2. 原始数据箱线图
ax2 = fig.add_subplot(2, 2, 2)
ax2.set_title('原始数据箱线图')
ax2.boxplot(original_data['value'], vert=False)
ax2.set_xlabel('value')
# 标记异常值
outliers_original = original_data[(original_data['value'] < lower_bound_iqr) | (original_data['value'] > upper_bound_iqr)]
ax2.scatter(outliers_original['value'], [1]*len(outliers_original), color='red', zorder=5, label='异常值')
ax2.legend()

# 3. 处理后数据分布
ax3 = fig.add_subplot(2, 2, 3)
ax3.set_title('异常值处理后数据分布')
ax3.hist(data['value'], bins=20, alpha=0.7, color='#99ff99', label='处理后分布')
ax3.axvline(mean_val, color='blue', linestyle='-', label=f'均值: {mean_val:.2f}')
ax3.legend()
ax3.set_ylabel('频数')
ax3.set_xlabel('value')

# 4. 处理后数据箱线图
ax4 = fig.add_subplot(2, 2, 4)
ax4.set_title('异常值处理后箱线图')
ax4.boxplot(data['value'], vert=False)
ax4.set_xlabel('value')

plt.tight_layout()
plt.savefig(save_path + '2.异常值处理_对比图.png')
plt.close()

# 绘制异常值位置对比
fig, ax = plt.subplots(figsize=(12, 6))
ax.set_title('异常值位置对比')
ax.scatter(range(len(original_data)), original_data['value'], alpha=0.7, color='blue', label='原始数据')
ax.scatter(range(len(data)), data['value'], alpha=0.7, color='green', label='处理后数据')
# 标记被替换的异常值
outlier_indices = data[data['is_outlier']].index
ax.scatter(outlier_indices, original_data.loc[outlier_indices, 'value'], color='red', s=100, label='被替换的异常值')
ax.legend()
ax.set_ylabel('value')
ax.set_xlabel('样本索引')
plt.tight_layout()
plt.savefig(save_path + '2.异常值处理_位置对比.png')
plt.close()

print("\n异常值检测结果：")
print(f"数据均值: {mean_val:.2f}")
print(f"数据标准差: {std_val:.2f}")
print(f"IQR: {IQR:.2f}")
print(f"\nIQR方法异常值范围: [{lower_bound_iqr:.2f}, {upper_bound_iqr:.2f}]")
print(f"3 Sigma方法异常值范围: [{lower_bound_sigma:.2f}, {upper_bound_sigma:.2f}]")
print(f"\nIQR方法检测到 {data['is_outlier_iqr'].sum()} 个异常值")
print(f"3 Sigma方法检测到 {data['is_outlier_sigma'].sum()} 个异常值")
print(f"总异常值数量: {data['is_outlier'].sum()}")

print("\n原始数据统计：")
print(original_data.describe())
print("\n异常值处理后数据统计：")
print(data[['value']].describe())
